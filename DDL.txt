CREATE TABLE Employees (
    ID Integer AUTO_INCREMENT PRIMARY KEY,
    Name VarChar(50),
    Email VarChar(50),
    Mobile VarChar(50),
    HireDate Date
);

CREATE TABLE Users (
    ID Integer AUTO_INCREMENT PRIMARY KEY,
    Name VarChar(50),
    Email VarChar(50) UNIQUE,
    Passwd VarChar(50)
);

CREATE TABLE StatusTypes (
    ID Integer AUTO_INCREMENT PRIMARY KEY,
    Type VarChar(50) UNIQUE
);

CREATE TABLE Attendance (
    Day Date,
    EmployeeID Integer,
    StatusTypeID Integer NOT NULL,
    WorkingHours Integer NOT NULL DEFAULT 0,
    
    PRIMARY KEY (EmployeeID, Day),
    FOREIGN KEY (EmployeeID) REFERENCES Employees (ID) ON UPDATE CASCADE ON DELETE CASCADE,
    FOREIGN KEY (StatusTypeID) REFERENCES StatusTypes (ID) ON UPDATE CASCADE ON DELETE CASCADE
);
